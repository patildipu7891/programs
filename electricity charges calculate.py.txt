#electricity charges calculate
u = int(input("Enter units: "))

if u <= 50:
    bill = u * 0.5
elif u <= 150:
    bill = 50 * 0.5 + (u - 50) * 0.75
elif u <= 250:
    bill = 50 * 0.5 + 100 * 0.75 + (u - 150) * 1.20
else:
    bill = 50 * 0.5 + 100 * 0.75 + 100 * 1.2 + (u - 250) * 1.20

bill = bill + bill * 0.2   

print("Electricity Bill =", bill)
